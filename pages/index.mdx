import { useRouter } from 'next/router'
import { useEffect } from 'react'

export default function Index() {
  const router = useRouter()

  useEffect(() => {
    router.replace('/networking/tcp-udp')
  }, [])

  return (
    <div style={{ padding: '2rem', textAlign: 'center' }}>
      <p>Redirection...</p>
    </div>
  )
}
